### Bài 1: Mảng cơ bản ⭐
Nhập mảng n phần tử. Tính min, max, trung bình, tổng. Không dùng STL.
CAU TRA LOI:
#include <iostream>
using namespace std;

int main() {
    int n;

    cout << "Nhap so phan tu: ";
    cin >> n;

    int a[1000];

    for (int i = 0; i < n; i++) {
        cout << "a[" << i << "] = ";
        cin >> a[i];
    }

    int minVal = a[0];
    int maxVal = a[0];
    int sum = 0;

    for (int i = 0; i < n; i++) {
        if (a[i] < minVal)
            minVal = a[i];

        if (a[i] > maxVal)
            maxVal = a[i];

        sum += a[i];
    }

    float avg = (float)sum / n;

    cout << "\nMin = " << minVal;
    cout << "\nMax = " << maxVal;
    cout << "\nTong = " << sum;
    cout << "\nTrung binh = " << avg;

    return 0;
}
### Bài 2: Mảng 2D ⭐⭐
Nhân 2 ma trận n×n. Tính định thức ma trận 3×3. Hiển thị đẹp.
CAU TRA LOI:
NHAN 2 NA TRAN
#include <iostream>
using namespace std;

int main() {
    int n;

    cout << "Nhap n: ";
    cin >> n;

    int A[100][100];
    int B[100][100];
    int C[100][100];

    cout << "\nNhap ma tran A:\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> A[i][j];
        }
    }

    cout << "\nNhap ma tran B:\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cin >> B[i][j];
        }
    }

    // Nhân ma trận
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {

            C[i][j] = 0;

            for (int k = 0; k < n; k++) {
                C[i][j] += A[i][k] * B[k][j];
            }
        }
    }

    cout << "\nMa tran ket qua:\n";

    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << C[i][j] << "\t";
        }
        cout << endl;
    }

    return 0;
}

### Bài 3: Con trỏ & cấp phát động ⭐⭐
Cài đặt mảng động tự resize (như `std::vector` đơn giản). Hỗ trợ push_back, pop_back, at(i).
CAU TRA LOI:
#include <iostream>
using namespace std;

class MyVector {
private:
    int* data;
    int size;
    int capacity;

    void resize(int newCapacity) {
        int* newData = new int[newCapacity];

        for (int i = 0; i < size; i++) {
            newData[i] = data[i];
        }

        delete[] data;

        data = newData;
        capacity = newCapacity;
    }

public:
    MyVector() {
        capacity = 2;
        size = 0;
        data = new int[capacity];
    }

    ~MyVector() {
        delete[] data;
    }

    void push_back(int value) {
        if (size == capacity) {
            resize(capacity * 2);
        }

        data[size] = value;
        size++;
    }

    void pop_back() {
        if (size > 0) {
            size--;
        }
    }

    int at(int index) {
        if (index < 0 || index >= size) {
            cout << "Index khong hop le!\n";
            return -1;
        }

        return data[index];
    }

    void print() {
        for (int i = 0; i < size; i++) {
            cout << data[i] << " ";
        }
        cout << endl;
    }

    int getSize() {
        return size;
    }
};

int main() {
    MyVector v;

    v.push_back(10);
    v.push_back(20);
    v.push_back(30);
    v.push_back(40);

    cout << "Mang sau push_back:\n";
    v.print();

    v.pop_back();

    cout << "Sau pop_back:\n";
    v.print();

    cout << "Phan tu tai index 1: ";
    cout << v.at(1);

    return 0;
}
### Bài 4: 🔥 Dự Án Mini — Student Score Manager ⭐⭐⭐
> **Cảm hứng:** BaiTapTongHop — Quản lý sinh viên (DSALab)

Xây dựng hệ thống quản lý điểm sinh viên bằng **mảng động**:
- Thêm / xóa / sửa sinh viên (tên, MSSV, điểm)
- Sắp xếp theo điểm (dùng Selection Sort hoặc Bubble Sort)
- Tìm kiếm theo tên hoặc MSSV (Linear Search)
- Thống kê: điểm cao nhất, thấp nhất, trung bình lớp
- Xuất danh sách ra file `diem_sinhvien.txt`
CAU TRA LOI:
#include <iostream>
#include <fstream>
#include <string>

using namespace std;

struct Student {
    string name;
    string id;
    float score;
};

class StudentManager {
private:
    Student* students;
    int size;
    int capacity;

    void resize(int newCapacity) {
        Student* newStudents = new Student[newCapacity];

        for (int i = 0; i < size; i++) {
            newStudents[i] = students[i];
        }

        delete[] students;
        students = newStudents;
        capacity = newCapacity;
    }

public:
    StudentManager() {
        size = 0;
        capacity = 2;
        students = new Student[capacity];
    }

    ~StudentManager() {
        delete[] students;
    }

    void addStudent() {
        if (size == capacity) {
            resize(capacity * 2);
        }

        cin.ignore();

        cout << "Nhap ten: ";
        getline(cin, students[size].name);

        cout << "Nhap MSSV: ";
        getline(cin, students[size].id);

        cout << "Nhap diem: ";
        cin >> students[size].score;

        size++;
    }

    void displayStudents() {
        cout << "\n===== DANH SACH SINH VIEN =====\n";

        for (int i = 0; i < size; i++) {
            cout << i + 1 << ". "
                 << students[i].name << " | "
                 << students[i].id << " | "
                 << students[i].score << endl;
        }
    }

    void deleteStudent(string id) {
        int pos = -1;

        for (int i = 0; i < size; i++) {
            if (students[i].id == id) {
                pos = i;
                break;
            }
        }

        if (pos == -1) {
            cout << "Khong tim thay!\n";
            return;
        }

        for (int i = pos; i < size - 1; i++) {
            students[i] = students[i + 1];
        }

        size--;

        cout << "Da xoa sinh vien.\n";
    }

    void editStudent(string id) {
        for (int i = 0; i < size; i++) {
            if (students[i].id == id) {

                cin.ignore();

                cout << "Nhap ten moi: ";
                getline(cin, students[i].name);

                cout << "Nhap diem moi: ";
                cin >> students[i].score;

                cout << "Cap nhat thanh cong!\n";
                return;
            }
        }

        cout << "Khong tim thay sinh vien.\n";
    }

    // Bubble Sort
    void sortByScore() {
        for (int i = 0; i < size - 1; i++) {
            for (int j = 0; j < size - i - 1; j++) {

                if (students[j].score > students[j + 1].score) {

                    Student temp = students[j];
                    students[j] = students[j + 1];
                    students[j + 1] = temp;
                }
            }
        }

        cout << "Da sap xep.\n";
    }

    void searchByName(string keyword) {
        bool found = false;

        for (int i = 0; i < size; i++) {

            if (students[i].name.find(keyword) != string::npos) {

                cout << students[i].name << " | "
                     << students[i].id << " | "
                     << students[i].score << endl;

                found = true;
            }
        }

        if (!found) {
            cout << "Khong tim thay.\n";
        }
    }

    void statistics() {
        if (size == 0) {
            cout << "Danh sach rong.\n";
            return;
        }

        float maxScore = students[0].score;
        float minScore = students[0].score;
        float sum = 0;

        for (int i = 0; i < size; i++) {

            if (students[i].score > maxScore)
                maxScore = students[i].score;

            if (students[i].score < minScore)
                minScore = students[i].score;

            sum += students[i].score;
        }

        cout << "\nDiem cao nhat: " << maxScore;
        cout << "\nDiem thap nhat: " << minScore;
        cout << "\nDiem trung binh: " << sum / size << endl;
    }

    void saveToFile() {
        ofstream file("diem_sinhvien.txt");

        for (int i = 0; i < size; i++) {
            file << students[i].name << " | "
                 << students[i].id << " | "
                 << students[i].score << endl;
        }

        file.close();

        cout << "Da luu file diem_sinhvien.txt\n";
    }
};

int main() {
    StudentManager sm;
    int choice;
    string id, keyword;

    do {
        cout << "\n===== MENU =====\n";
        cout << "1. Them sinh vien\n";
        cout << "2. Hien thi danh sach\n";
        cout << "3. Xoa sinh vien\n";
        cout << "4. Sua sinh vien\n";
        cout << "5. Sap xep theo diem\n";
        cout << "6. Tim kiem theo ten\n";
        cout << "7. Thong ke diem\n";
        cout << "8. Luu file\n";
        cout << "0. Thoat\n";

        cout << "Chon: ";
        cin >> choice;

        switch (choice) {

        case 1:
            sm.addStudent();
            break;

        case 2:
            sm.displayStudents();
            break;

        case 3:
            cin.ignore();
            cout << "Nhap MSSV can xoa: ";
            getline(cin, id);
            sm.deleteStudent(id);
            break;

        case 4:
            cin.ignore();
            cout << "Nhap MSSV can sua: ";
            getline(cin, id);
            sm.editStudent(id);
            break;

        case 5:
            sm.sortByScore();
            break;

        case 6:
            cin.ignore();
            cout << "Nhap ten can tim: ";
            getline(cin, keyword);
            sm.searchByName(keyword);
            break;

        case 7:
            sm.statistics();
            break;

        case 8:
            sm.saveToFile();
            break;

        case 0:
            cout << "Thoat chuong trinh.\n";
            break;

        default:
            cout << "Lua chon khong hop le.\n";
        }

    } while (choice != 0);

    return 0;
}
