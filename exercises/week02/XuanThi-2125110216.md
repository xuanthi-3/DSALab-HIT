### Bài 1: Mảng cơ bản ⭐
Nhập mảng n phần tử. Tính min, max, trung bình, tổng. Không dùng STL.
CÂU TRA LOI:
#include <iostream>
using namespace std;

int main() {
    int n;

    cout << "Nhap so phan tu: ";
    cin >> n;

    int a[1000];

    for (int i = 0; i < n; i++) {
        cout << "a[" << i << "] = ";
        cin >> a[i];
    }

    int min = a[0];
    int max = a[0];
    int tong = 0;

    for (int i = 0; i < n; i++) {
        if (a[i] < min)
            min = a[i];

        if (a[i] > max)
            max = a[i];

        tong += a[i];
    }

    float trungBinh = (float)tong / n;

    cout << "\nMin = " << min;
    cout << "\nMax = " << max;
    cout << "\nTong = " << tong;
    cout << "\nTrung binh = " << trungBinh;

    return 0;
}
### Bài 2: Mảng 2D ⭐⭐
Nhân 2 ma trận n×n. Tính định thức ma trận 3×3. Hiển thị đẹp.
CAU TRA LOI:
#include <iostream>
#include <iomanip>
using namespace std;

int main() {
    int n;

    cout << "Nhap kich thuoc ma tran n x n: ";
    cin >> n;

    int a[10][10];

    cout << "\nNhap ma tran:\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << "a[" << i << "][" << j << "] = ";
            cin >> a[i][j];
        }
    }

    cout << "\nMa tran vua nhap:\n";
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            cout << setw(5) << a[i][j];
        }
        cout << endl;
    }

    // Tính định thức ma trận 3x3
    if (n == 3) {
        int det =
            a[0][0] * (a[1][1] * a[2][2] - a[1][2] * a[2][1])
          - a[0][1] * (a[1][0] * a[2][2] - a[1][2] * a[2][0])
          + a[0][2] * (a[1][0] * a[2][1] - a[1][1] * a[2][0]);

        cout << "\nDinh thuc ma tran = " << det;
    }
    else {
        cout << "\nChi tinh dinh thuc cho ma tran 3x3.";
    }

    return 0;
}
### Bài 3: Con trỏ & cấp phát động ⭐⭐
Cài đặt mảng động tự resize (như `std::vector` đơn giản). Hỗ trợ push_back, pop_back, at(i).
CAU TRA LOI:
#include <iostream>
using namespace std;

class DynamicArray {
private:
    int* data;
    int size;
    int capacity;

    void resize() {
        capacity *= 2;

        int* newData = new int[capacity];

        for (int i = 0; i < size; i++) {
            newData[i] = data[i];
        }

        delete[] data;
        data = newData;
    }

public:
    DynamicArray() {
        capacity = 2;
        size = 0;
        data = new int[capacity];
    }

    ~DynamicArray() {
        delete[] data;
    }

    void push_back(int value) {
        if (size == capacity) {
            resize();
        }

        data[size] = value;
        size++;
    }

    void pop_back() {
        if (size > 0) {
            size--;
        }
    }

    int at(int index) {
        if (index >= 0 && index < size) {
            return data[index];
        }

        cout << "Index khong hop le!\n";
        return -1;
    }

    void display() {
        cout << "Mang: ";

        for (int i = 0; i < size; i++) {
            cout << data[i] << " ";
        }

        cout << endl;
    }
};

int main() {
    DynamicArray arr;

    arr.push_back(10);
    arr.push_back(20);
    arr.push_back(30);

    arr.display();

    arr.pop_back();

    arr.display();

    cout << "Phan tu tai vi tri 0: " << arr.at(0);

    return 0;
}
### Bài 4: 🔥 Dự Án Mini — Student Score Manager ⭐⭐⭐
> **Cảm hứng:** BaiTapTongHop — Quản lý sinh viên (DSALab)

Xây dựng hệ thống quản lý điểm sinh viên bằng **mảng động**:
- Thêm / xóa / sửa sinh viên (tên, MSSV, điểm)
- Sắp xếp theo điểm (dùng Selection Sort hoặc Bubble Sort)
- Tìm kiếm theo tên hoặc MSSV (Linear Search)
- Thống kê: điểm cao nhất, thấp nhất, trung bình lớp
- Xuất danh sách ra file `diem_sinhvien.txt`
CAU TRA LOI:
#include <iostream>
#include <fstream>
#include <string>
using namespace std;

struct Student {
    string ten;
    string mssv;
    float diem;
};

void themSinhVien(Student ds[], int& n) {
    cout << "\nNhap ten: ";
    cin.ignore();
    getline(cin, ds[n].ten);

    cout << "Nhap MSSV: ";
    getline(cin, ds[n].mssv);

    cout << "Nhap diem: ";
    cin >> ds[n].diem;

    n++;

    cout << "Them sinh vien thanh cong!\n";
}

void xoaSinhVien(Student ds[], int& n, string mssv) {
    int pos = -1;

    for (int i = 0; i < n; i++) {
        if (ds[i].mssv == mssv) {
            pos = i;
            break;
        }
    }

    if (pos == -1) {
        cout << "Khong tim thay sinh vien!\n";
        return;
    }

    for (int i = pos; i < n - 1; i++) {
        ds[i] = ds[i + 1];
    }

    n--;

    cout << "Xoa thanh cong!\n";
}

void timKiem(Student ds[], int n, string key) {
    bool found = false;

    for (int i = 0; i < n; i++) {
        if (ds[i].ten == key || ds[i].mssv == key) {
            cout << "\nTim thay:\n";
            cout << "Ten: " << ds[i].ten << endl;
            cout << "MSSV: " << ds[i].mssv << endl;
            cout << "Diem: " << ds[i].diem << endl;

            found = true;
        }
    }

    if (!found) {
        cout << "Khong tim thay!\n";
    }
}

void sapXep(Student ds[], int n) {
    for (int i = 0; i < n - 1; i++) {
        for (int j = i + 1; j < n; j++) {
            if (ds[i].diem < ds[j].diem) {
                Student temp = ds[i];
                ds[i] = ds[j];
                ds[j] = temp;
            }
        }
    }

    cout << "Da sap xep giam dan theo diem!\n";
}

void thongKe(Student ds[], int n) {
    if (n == 0) {
        cout << "Danh sach rong!\n";
        return;
    }

    float max = ds[0].diem;
    float min = ds[0].diem;
    float tong = 0;

    for (int i = 0; i < n; i++) {
        if (ds[i].diem > max)
            max = ds[i].diem;

        if (ds[i].diem < min)
            min = ds[i].diem;

        tong += ds[i].diem;
    }

    cout << "\nDiem cao nhat: " << max;
    cout << "\nDiem thap nhat: " << min;
    cout << "\nDiem trung binh: " << tong / n << endl;
}

void xuatFile(Student ds[], int n) {
    ofstream file("diem_sinhvien.txt");

    for (int i = 0; i < n; i++) {
        file << ds[i].ten << " | "
             << ds[i].mssv << " | "
             << ds[i].diem << endl;
    }

    file.close();

    cout << "Da xuat file diem_sinhvien.txt\n";
}

void hienThi(Student ds[], int n) {
    cout << "\n===== DANH SACH SINH VIEN =====\n";

    for (int i = 0; i < n; i++) {
        cout << "\nSinh vien " << i + 1 << endl;
        cout << "Ten : " << ds[i].ten << endl;
        cout << "MSSV: " << ds[i].mssv << endl;
        cout << "Diem: " << ds[i].diem << endl;
    }
}

int main() {
    Student ds[100];
    int n = 0;

    int choice;

    do {
        cout << "\n=== QUAN LY DIEM SINH VIEN ===\n";
        cout << "1. Them sinh vien\n";
        cout << "2. Xoa sinh vien\n";
        cout << "3. Tim kiem\n";
        cout << "4. Xep hang lop\n";
        cout << "5. Xuat bao cao\n";
        cout << "6. Hien thi danh sach\n";
        cout << "0. Thoat\n";

        cout << "Nhap lua chon: ";
        cin >> choice;

        switch (choice) {
        case 1:
            themSinhVien(ds, n);
            break;

        case 2: {
            string mssv;
            cout << "Nhap MSSV can xoa: ";
            cin >> mssv;
            xoaSinhVien(ds, n, mssv);
            break;
        }

        case 3: {
            string key;
            cout << "Nhap ten hoac MSSV: ";
            cin.ignore();
            getline(cin, key);
            timKiem(ds, n, key);
            break;
        }

        case 4:
            sapXep(ds, n);
            break;

        case 5:
            thongKe(ds, n);
            xuatFile(ds, n);
            break;

        case 6:
            hienThi(ds, n);
            break;

        case 0:
            cout << "Thoat chuong trinh!\n";
            break;

        default:
            cout << "Lua chon khong hop le!\n";
        }

    } while (choice != 0);

    return 0;
}
