### Bài 1: Linear Search ⭐
Tìm kiếm tuyến tính trên mảng số nguyên và mảng chuỗi. Đếm số bước so sánh.
số nguyên
#include <iostream>
using namespace std;

int linearSearch(int arr[], int n, int x, int &steps)
{
    steps = 0;

    for (int i = 0; i < n; i++)
    {
        steps++;

        if (arr[i] == x)
        {
            return i;
        }
    }

    return -1;
}

int main()
{
    int arr[] = {4, 7, 1, 9, 3};
    int n = 5;

    int x;
    cout << "Nhap so can tim: ";
    cin >> x;

    int steps;

    int pos = linearSearch(arr, n, x, steps);

    if (pos != -1)
    {
        cout << "Tim thay tai vi tri: "
             << pos << endl;
    }
    else
    {
        cout << "Khong tim thay\n";

         << steps << endl;

    return 0;
}

### Bài 2: Binary Search ⭐⭐
Cài đặt Binary Search iterative + recursive. Tìm vị trí đầu tiên và cuối cùng của phần tử trùng.
Binary Search Iterative
#include <iostream>
using namespace std;

int binarySearch(int arr[], int n, int x)
{
    int left = 0;
    int right = n - 1;

    while (left <= right)
    {
        int mid = (left + right) / 2;

        if (arr[mid] == x)
        {
            return mid;
        }

        if (arr[mid] < x)
        {
            left = mid + 1;
        }
        else
        {
            right = mid - 1;
        }
    }

    return -1;
}

int main()
{
    int arr[] = {1,2,3,4,5,6,7,8,9};
    int n = 9;

    int x;
    cin >> x;

    int pos = binarySearch(arr, n, x);

    cout << "Vi tri: " << pos;

    return 0;
}
Binary Search Recursive
#include <iostream>
using namespace std;

int binaryRecursive(
    int arr[],
    int left,
    int right,
    int x)
{
    if (left > right)
    {
        return -1;
    }

    int mid = (left + right) / 2;

    if (arr[mid] == x)
    {
        return mid;
    }

    if (arr[mid] < x)
    {
        return binaryRecursive(
            arr,
            mid + 1,
            right,
            x);
    }

    return binaryRecursive(
        arr,
        left,
        mid - 1,
        x);
}

int main()
{
    int arr[] = {1,2,3,4,5,6,7,8,9};

    int n = 9;

    int x;
    cin >> x;

    cout << binaryRecursive(arr, 0, n - 1, x);

    return 0;
}
First Occurrence
int firstOccurrence(int arr[], int n, int x)
{
    int left = 0;
    int right = n - 1;

    int ans = -1;

    while (left <= right)
    {
        int mid = (left + right) / 2;

        if (arr[mid] == x)
        {
            ans = mid;
            right = mid - 1;
        }
        else if (arr[mid] < x)
        {
            left = mid + 1;
        }
        else
        {
            right = mid - 1;
        }
    }

    return ans;
}

### Bài 3: So sánh hiệu năng ⭐⭐
Đo thời gian tìm kiếm với n = 10.000, 100.000, 1.000.000 phần tử. Vẽ bảng so sánh.
#include <iostream>
#include <chrono>

using namespace std;
using namespace chrono;

int linearSearch(int arr[], int n, int x)
{
    for (int i = 0; i < n; i++)
    {
        if (arr[i] == x)
        {
            return i;
        }
    }

    return -1;
}

int binarySearch(int arr[], int n, int x)
{
    int left = 0;
    int right = n - 1;

    while (left <= right)
    {
        int mid = (left + right) / 2;

        if (arr[mid] == x)
        {
            return mid;
        }

        if (arr[mid] < x)
        {
            left = mid + 1;
        }
        else
        {
            right = mid - 1;
        }
    }

    return -1;
}

void benchmark(int n)
{
    int* arr = new int[n];

    for (int i = 0; i < n; i++)
    {
        arr[i] = i;
    }

    int target = n - 1;

    auto start =
    high_resolution_clock::now();

    linearSearch(arr, n, target);

    auto end =
    high_resolution_clock::now();

    auto linearTime =
    duration_cast<microseconds>
    (end - start);

    start =
    high_resolution_clock::now();

    binarySearch(arr, n, target);

    end =
    high_resolution_clock::now();

    auto binaryTime =
    duration_cast<microseconds>
    (end - start);

    cout << n
         << "\t"
         << linearTime.count()
         << "us\t"
         << binaryTime.count()
         << "us\n";

    delete[] arr;
}

int main()
{
    cout << "n\tLinear\tBinary\n";

    benchmark(10000);
    benchmark(100000);
    benchmark(1000000);

    return 0;
}
### Bài 4: 🔥 Dự Án Mini — Smart Search Engine ⭐⭐⭐
> **Cảm hứng:** [Brute Force Search — algorithm-visualizer](https://algorithm-visualizer.org)
#include <iostream>
#include <string>
#include <chrono>

using namespace std;
using namespace chrono;

struct Contact
{
    string name;
    string phone;
};

Contact contacts[100] =
{
    {"Nguyen Van Minh", "0901234567"},
    {"Tran Thi Minh Anh", "0912345678"},
    {"Le Minh Tuan", "0923456789"},
    {"Nguyen Van An", "0934567890"},
    {"Pham Gia Bao", "0945678901"}
};

int total = 5;

bool contains(string s, string sub)
{
    return s.find(sub) != string::npos;
}

void searchByName(string keyword)
{
    int steps = 0;

    auto start =
    high_resolution_clock::now();

    bool found = false;

    for (int i = 0; i < total; i++)
    {
        steps++;

        if (contains(contacts[i].name, keyword))
        {
            if (!found)
            {
                cout << "\nTim thay:\n";
            }

            found = true;

            cout << contacts[i].name
                 << " - "
                 << contacts[i].phone
                 << endl;
        }
    }

    auto end =
    high_resolution_clock::now();

    auto duration =
    duration_cast<microseconds>
    (end - start);

    if (!found)
    {
        cout << "Khong tim thay\n";
    }

    cout << "So buoc so sanh: "
         << steps << endl;

    cout << "Thoi gian: "
         << duration.count()
         << " us\n";
}

int binarySearchPhone(
    string phone,
    int &steps)
{
    int left = 0;
    int right = total - 1;

    while (left <= right)
    {
        steps++;

        int mid = (left + right) / 2;

        if (contacts[mid].phone == phone)
        {
            return mid;
        }

        if (contacts[mid].phone < phone)
        {
            left = mid + 1;
        }
        else
        {
            right = mid - 1;
        }
    }

    return -1;
}

void searchByPhone(string phone)
{
    int steps = 0;

    auto start =
    high_resolution_clock::now();

    int pos =
    binarySearchPhone(phone, steps);

    auto end =
    high_resolution_clock::now();

    auto duration =
    duration_cast<microseconds>
    (end - start);

    if (pos != -1)
    {
        cout << contacts[pos].name
             << " - "
             << contacts[pos].phone
             << endl;
    }
    else
    {
        cout << "Khong tim thay\n";
    }

    cout << "So buoc: "
         << steps << endl;

    cout << "Thoi gian: "
         << duration.count()
         << " us\n";
}

void suggestNames()
{
    cout << "\nGoi y:\n";

    for (int i = 0; i < 3 && i < total; i++)
    {
        cout << contacts[i].name << endl;
    }
}

int main()
{
    int choice;

    do
    {
        cout << "\n===== SMART SEARCH =====\n";
        cout << "1. Tim theo ten\n";
        cout << "2. Tim theo SDT\n";
        cout << "0. Thoat\n";

        cin >> choice;

        cin.ignore();

        if (choice == 1)
        {
            string keyword;

            cout << "Nhap ten can tim: ";
            getline(cin, keyword);

            searchByName(keyword);
        }
        else if (choice == 2)
        {
            string phone;

            cout << "Nhap SDT: ";
            getline(cin, phone);

            searchByPhone(phone);
        }

    } while (choice != 0);

    return 0;
}
Xây dựng hệ thống tìm kiếm danh bạ điện thoại:
- **Tìm theo tên:** Linear Search (hỗ trợ tìm kiếm mờ — chứa chuỗi con)
- **Tìm theo số điện thoại:** Binary Search (sau khi sort theo SĐT)
- **Thống kê:** hiển thị số bước so sánh, thời gian tìm kiếm
- **Gợi ý:** nếu không tìm thấy, gợi ý 3 tên gần giống nhất
